103 float AL_Lux(unsigned int ch0, unsigned int ch1)
104 {
105  float k = ch1/ch0;

107 
108  if((k>=0)&& (k<=0.52))
109  Lux=(0.0315*ch0)-(0.0593*ch0*pow(k,1.4));

111  Lux=(0.0229*ch0)-(0.0291*ch1);

116  else
117  Lux=0;
